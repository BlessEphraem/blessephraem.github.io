interface Pagination {

  data:    T;
  meta:    Pagination;

  Idle    = 'idle',
  Loading = 'loading',
  Done    = 'done',
  Error   = 'error',

// ── Generic function ──────────────────────────────────────────────────────────
async function apiFetch<T>(
  endpoint: string,
  options?: RequestInit,
): Promise<ApiResponse<T>> {
  const res  = await fetch(endpoint, options);
  const body = (await res.json()) as ApiResponse<T>;

  if (!res.ok) {
    throw new Error(body.error ?? `Request failed: ${res.status}`);
  }

  return body;
}

// ── Type alias / union / intersection ────────────────────────────────────────

  set(entity: T): void {
    this.store.set(entity.id, entity);
  }

  get(id: ID): T | undefined {
    return this.store.get(id);
  }

  getAll(): T[] {
    return [...this.store.values()];
  }
}

```python
from __future__ import annotations

import asyncio
import functools
from dataclasses import dataclass, field
from typing import Generic, TypeVar, Iterator

T = TypeVar("T")

# ── Triple-quoted docstring (triple-quoted-string) ─────────────────────────────
def fibonacci(n: int) -> int:
    """
    Return the n-th Fibonacci number.

    Args:
        n: Position in the sequence (0-indexed).

    Returns:
        The Fibonacci number at position ``n``.
    """
    if n < 2:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)


# ── Decorator ──────────────────────────────────────────────────────────────────
def retry(times: int = 3):
    """Decorator that retries a coroutine on exception."""
    def decorator(fn):
        @functools.wraps(fn)
        async def wrapper(*args, **kwargs):
            for attempt in range(1, times + 1):
                try:
                    return await fn(*args, **kwargs)
                except Exception as exc:          # noqa: BLE001
                    if attempt == times:
                        raise
                    print(f"[retry] attempt {attempt} failed: {exc}")
        return wrapper
    return decorator


# ── Dataclass / class with self / super ───────────────────────────────────────
@dataclass
class Stack(Generic[T]):
    """Generic LIFO stack."""
    _items: list[T] = field(default_factory=list)

    def push(self, item: T) -> None:
        self._items.append(item)

    def pop(self) -> T:
        if not self._items:
            raise IndexError("pop from empty stack")
        return self._items.pop()

    def __len__(self) -> int:
        return len(self._items)

    def __iter__(self) -> Iterator[T]:
        return reversed(self._items)


@dataclass
class BoundedStack(Stack[T]):
    capacity: int = 64

    def push(self, item: T) -> None:
        if len(self) >= self.capacity:
            raise OverflowError(f"Stack full ({self.capacity})")
        super().push(item)


# ── Async / control-flow ──────────────────────────────────────────────────────
@retry(times=2)
async def fetch_data(url: str) -> bytes:
    reader, writer = await asyncio.open_connection("example.com", 443)
    writer.write(f"GET {url} HTTP/1.0\r\n\r\n".encode())
    data = await reader.read(4096)
    writer.close()
    return data
```

### Bash — variables, functions, control flow

Covers: `comment`, `function`, `variable`, `string`, `keyword`, `builtin`.

```bash
#!/usr/bin/env bash
# Deploy script — builds and pushes the wiki to GitHub Pages

set -euo pipefail

# ── Variables ─────────────────────────────────────────────────────────────────
REPO_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
BUILD_DIR="${REPO_DIR}/build"
BRANCH="gh-pages"
REMOTE="${1:-origin}"

# ── Functions ─────────────────────────────────────────────────────────────────
log()  { echo "[$(date '+%H:%M:%S')] $*"; }
die()  { echo "ERROR: $*" >&2; exit 1; }

check_deps() {
  for cmd in git node npm; do
    command -v "$cmd" &>/dev/null || die "missing dependency: $cmd"
  done
}

build() {
  log "Building site…"
  cd "$REPO_DIR"
  npm ci --prefer-offline
  npm run build

  [[ -d "$BUILD_DIR" ]] || die "build output not found at $BUILD_DIR"
  log "Build complete → $BUILD_DIR"
}

deploy() {
  local commit_msg="deploy: $(git rev-parse --short HEAD) [$(date -u '+%Y-%m-%dT%H:%M:%SZ')]"

  log "Deploying to $REMOTE/$BRANCH…"
  git -C "$BUILD_DIR" init
  git -C "$BUILD_DIR" checkout -b "$BRANCH"
  git -C "$BUILD_DIR" add -A
  git -C "$BUILD_DIR" commit -m "$commit_msg"
  git -C "$BUILD_DIR" push -f "$REMOTE" "$BRANCH"
  log "Done."
}

# ── Entry point ───────────────────────────────────────────────────────────────
check_deps
build
deploy
```

### PowerShell — variables, cmdlets, strings, control flow

Covers: `comment`, `variable`, `function`, `string`, `keyword`, `operator`.

```powershell
# Build & deploy script for Windows
#Requires -Version 7.0

Set-StrictMode -Version Latest
$ErrorActionPreference = 'Stop'

# ── Config ─────────────────────────────────────────────────────────────────────
$RepoDir  = $PSScriptRoot
$BuildDir = Join-Path $RepoDir 'build'
$Branch   = 'gh-pages'
$Remote   = if ($args.Count -gt 0) { $args[0] } else { 'origin' }

# ── Functions ──────────────────────────────────────────────────────────────────
function Write-Log {
  param([string]$Message)
  $ts = Get-Date -Format 'HH:mm:ss'
  Write-Host "[$ts] $Message" -ForegroundColor Cyan
}

function Assert-Command {
  param([string[]]$Commands)
  foreach ($cmd in $Commands) {
    if (-not (Get-Command $cmd -ErrorAction SilentlyContinue)) {
      throw "Missing dependency: $cmd"
    }
  }
}

function Invoke-Build {
  Write-Log 'Building site…'
  Set-Location $RepoDir
  npm ci --prefer-offline
  npm run build

  if (-not (Test-Path $BuildDir)) {
    throw "Build output not found at $BuildDir"
  }
  Write-Log "Build complete → $BuildDir"
}

function Invoke-Deploy {
  $hash = git rev-parse --short HEAD
  $date = (Get-Date -Format 'yyyy-MM-ddTHH:mm:ssZ')
  $msg  = "deploy: $hash [$date]"

  Write-Log "Deploying to $Remote/$Branch…"
  Push-Location $BuildDir
  git init
  git checkout -b $Branch
  git add -A
  git commit -m $msg
  git push -f $Remote $Branch
  Pop-Location
  Write-Log 'Done.'
}

# ── Entry point ────────────────────────────────────────────────────────────────
Assert-Command 'git', 'node', 'npm'
Invoke-Build
Invoke-Deploy
```
